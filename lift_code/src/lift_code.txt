#include <Wire.h>
#include <Arduino.h>
#include <TFT_eSPI.h>  
 
TFT_eSPI tft = TFT_eSPI();
 
#define MMA8452Q_ADDR 0x1D
 
long count = 0;
// Registers
#define WHO_AM_I 0x0D
#define CTRL_REG1 0x2A
#define OUT_X_MSB 0x01
 
#define AIN1 1
#define AIN2 2
#define PWMA 3
#define PWMB 12
#define STBY 16
#define BIN1 10
#define BIN2 11
 
#define BTN_FORWARD 0
#define BTN_BACKWARD 14
 
 
void writeRegister(byte reg, byte value) {
  Wire.beginTransmission(MMA8452Q_ADDR);
  Wire.write(reg);
  Wire.write(value);
  Wire.endTransmission();
}
 
byte readRegister(byte reg) {
  Wire.beginTransmission(MMA8452Q_ADDR);
  Wire.write(reg);
  Wire.endTransmission(false);
  Wire.requestFrom(MMA8452Q_ADDR, 1);
  return Wire.read();
}
 
void readAccel(int16_t &x, int16_t &y, int16_t &z) {
  Wire.beginTransmission(MMA8452Q_ADDR);
  Wire.write(OUT_X_MSB);
  Wire.endTransmission(false);
  Wire.requestFrom(MMA8452Q_ADDR, 6);
 
  int16_t rawX = (Wire.read() << 8) | Wire.read();
  int16_t rawY = (Wire.read() << 8) | Wire.read();
  int16_t rawZ = (Wire.read() << 8) | Wire.read();
 
  x = rawX >> 4;
  y = rawY >> 4;
  z = rawZ >> 4;
}
 
// lift going down
void moveForward(int left_power, int right_power) {
  tft.setTextColor(TFT_WHITE, TFT_BLACK);
  digitalWrite(AIN1, HIGH);
  digitalWrite(AIN2, LOW);
  digitalWrite(BIN1, HIGH);
  digitalWrite(BIN2, LOW);
  analogWrite(PWMA, left_power);
  analogWrite(PWMB, right_power);
}
 
// lift going up
void moveBackward(int left_power, int right_power) {
  tft.setTextColor(TFT_WHITE, TFT_BLACK);
  digitalWrite(AIN1, LOW);
  digitalWrite(AIN2, HIGH);
  digitalWrite(BIN1, LOW);
  digitalWrite(BIN2, HIGH);
  analogWrite(PWMA, left_power);
  analogWrite(PWMB, right_power);
}
 
void stopMotor() {
  digitalWrite(AIN1, LOW);
  digitalWrite(AIN2, LOW);
  digitalWrite(BIN1, LOW);
  digitalWrite(BIN2, LOW);
  digitalWrite(PWMA, LOW);
  digitalWrite(PWMB, LOW);
}
 
void setup() {
  Wire.begin(17, 18);
 
  // Init screen
  tft.init();
  tft.setRotation(1);
  tft.fillScreen(TFT_BLACK);
  tft.setTextColor(TFT_GREEN, TFT_BLACK);
  tft.setTextSize(2);
 
  pinMode(AIN1, OUTPUT);
  pinMode(AIN2, OUTPUT);
  pinMode(BIN1, OUTPUT);
  pinMode(BIN2, OUTPUT);
  pinMode(PWMA, OUTPUT);
  pinMode(PWMB, OUTPUT);
  pinMode(STBY, OUTPUT);
  digitalWrite(STBY, HIGH);
 
  pinMode(BTN_FORWARD, INPUT_PULLUP);
  pinMode(BTN_BACKWARD, INPUT_PULLUP);
 
  digitalWrite(AIN1, LOW);
  digitalWrite(AIN2, LOW);
  digitalWrite(PWMA, LOW);
  digitalWrite(PWMB, LOW);
 
  tft.setCursor(0, 0);
  tft.println("Starting...");
 
  delay(500);
 
  // Check sensor
  byte id = readRegister(WHO_AM_I);
 
  tft.print("WHO_AM_I: 0x");
  tft.println(id, HEX);
 
  if (id != 0x2A) {
    tft.println("Sensor NOT found!");
    while (1);
  }
 
  // Configure sensor
  writeRegister(CTRL_REG1, 0x00);
  writeRegister(CTRL_REG1, 0x01);
 
  tft.println("MMA8452Q ready");
  delay(1000);
}
 
void loop() {
  
  int16_t x, y, z;
  readAccel(x, y, z);
 
  float x_g = x / 1024.0;
  float y_g = y / 1024.0;
  float z_g = z / 1024.0;
  
  float pitch = atan2(x_g, sqrt(y_g * y_g + z_g * z_g)) * 180.0 / PI;
  float roll  = atan2(y_g, sqrt(x_g * x_g + z_g * z_g)) * 180.0 / PI;
 
  tft.fillRect(0, 40, 240, 100, TFT_BLACK);
 
  tft.setCursor(0, 40);
  tft.print("Roll: ");
  tft.println(roll, 1);
 
  tft.print("Pitch: ");
  tft.println(pitch, 1);
 
  tft.print("Z: ");
  tft.println(z);
 
  long power = map(abs(roll), 0, 90, 0, 255);
  long power_down = map(abs(roll), 0, 90, 0, 150);
 
  bool fwdPressed = digitalRead(BTN_FORWARD) == LOW;
  bool bwdPressed = digitalRead(BTN_BACKWARD) == LOW;
 
    if (count < 150) {
      if (roll < -5){
        moveForward(150, power_down);
      }
      else if (roll > 5){
        moveForward(power_down, 150);
      }
      else{
        moveForward(150, 150);
      }
    }
 
    else if (count > 200 && count < 350) {
 
      if (roll < -5){
        moveBackward(power, 255);
      }
      else if (roll > 5){
        moveBackward(255, power);
      }
      else{
        moveBackward(255, 255);
      }
    } else {
      stopMotor();
        }
 
    delay(100);
    count++;
}
